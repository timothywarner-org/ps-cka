    Exists -->|"No"| Create["Create cluster"]
    Create --> CreateOK{Success?}
    CreateOK -->|"No"| Fail3(["EXIT: Creation failed"]):::fail
    CreateOK -->|"Yes"| IsWork{Workloads\ntopology?}

    IsWork -->|"No"| Step4

    TutCheck -->|"No"| Done(["CKA Lab Ready"]):::done
    TutCheck -->|"Yes"| Step6["Step 6: Run Tutorial\ninteractive, on-rails"]:::tut
    Step6 --> Done
