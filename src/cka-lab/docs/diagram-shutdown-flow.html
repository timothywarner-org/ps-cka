kind-down.ps1 — Cluster Shutdown Flow

Interactive shutdown mode selection, cluster deletion, optional Docker Desktop and WSL2 teardown.

flowchart TD
    Start([".\kind-down.ps1"]):::entry --> Source["Source lib/CkaLab.ps1\nInitialize-LabPath"]:::lib
    Source --> Banner["Banner: KIND Cluster Shutdown"]

    Banner --> ShutMenu{"Shutdown Mode\n-StopDockerDesktop skips"}:::menu
    ShutMenu -->|"1 Cluster only"| ClusterOnly["DD stays running\nFast re-creation"]
    ShutMenu -->|"2 Full shutdown"| FullShut["Stop DD + WSL2"]

    ClusterOnly --> PruneMenu
    FullShut --> PruneMenu

    PruneMenu{"Prune Docker\nimages? y/N"}:::menu -->|"Yes"| DoPrune["Will prune"]
    PruneMenu -->|"No"| NoPrune[" "]:::invisible

    DoPrune --> Step1
    NoPrune --> Step1

    Step1["Step 1: Delete KIND Cluster"]:::step --> DockerUp{Docker\nrunning?}
    DockerUp -->|"No"| Gone["Containers already gone"]
    DockerUp -->|"Yes"| ClusterCheck{Cluster\nexists?}
    ClusterCheck -->|"No"| NoDel["Nothing to delete"]
    ClusterCheck -->|"Yes"| Delete["kind delete cluster\n--name cka-lab"]
    Delete --> DelOK{Success?}
    DelOK -->|"Yes"| Deleted["Cluster deleted"]:::done
    DelOK -->|"No"| ForceRM["Force-remove containers\nby label filter"]:::warn

    Gone --> Step2
    NoDel --> Step2
    Deleted --> Step2
    ForceRM --> Step2

    Step2{"Step 2: Prune?"}:::step -->|"Yes"| Prune["docker system prune -af\n--volumes"]
    Step2 -->|"No"| SkipPrune["Skip prune"]

    Prune --> Step3
    SkipPrune --> Step3

    Step3{"Step 3: Stop DD?"}:::step -->|"Full shutdown"| StopDD["Stop-DockerDesktop\nKill DD + backend + proxy"]:::shutdown
    Step3 -->|"Cluster only"| SkipDD["DD stays running"]

    StopDD --> Step4
    SkipDD --> Step4

    Step4{"Step 4: WSL2?"}:::step -->|"Full shutdown"| WSLPend["WSL shutdown pending\nDeferred to end"]:::shutdown
    Step4 -->|"Cluster only"| SkipWSL["Skip WSL"]

    WSLPend --> Step5
    SkipWSL --> Step5

    Step5["Step 5: Verification\nTest-DockerReady\nWrite-HostMemory"]:::step --> FinalBanner

    FinalBanner{"Shutdown\nmode?"}
    FinalBanner -->|"Cluster only"| DoneFast(["CKA Lab Torn Down\nDD running, ready for kind-up"]):::done
    FinalBanner -->|"Full"| WSL["wsl --shutdown\n(3s delay, last action)"]:::shutdown
    WSL --> DoneFull(["CKA Lab Shut Down\nDD stopped, WSL down"]):::done
